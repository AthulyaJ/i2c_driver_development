//testing code for i2cMasterSend API
#include "i2C_driver.h"
#define SLAVE_ADDRS  0x62		//address of the slave
I2C_Handle_t I2C1Handle;

//some data to transmit
uint8_t data[] = "testing I2C master transmit \n";

//GPIO initialization
Void I2C_GPIO_Inits(void)
{
  GPIO_Handle_t I2CPins;
  I2CPins.pGPIO = GPIOB;
  I2CPins.GPIO_Pin_Config.GPIO_Pin_Mode = GPIO_MODE_ALTFNCTN;
  I2CPins.GPIO_Pin_Config.GPIO_Pin_Output_Type = GPIO_OP_TYPE_OD; //output type is as open drain
  I2CPins.GPIO_Pin_Config.GPIO_Pin_PuPd_Control = GPIO_PIN_PU;    //pull up
  I2CPins.GPIO_Pin_Config.GPIO_Pin_ALT_Mode = 4;
  I2CPins.GPIO_Pin_Config.GPIO_Pin_Speed = GPIO_SPEED_FAST;

  //SCL
  I2CPins.GPIO_Pin_Config.GPIO_Pin_Number = GPIO_PIN_NO_6; ///select the pin accordingly
  GPIO_Inits(&I2CPins);
  
  // SDA
  I2CPins.GPIO_Pin_Config.GPIO_Pin_Number = GPIO_PIN_NO_9; ///select the pin accordingly
  GPIO_Inits(&I2CPins);
}
void I2C1_Inits(void)
{
  I2C1Handle.pI2C = I2C1;
  I2C1Handle.I2C_Config_t.I2CACKControl = I2C_ACK_ENABLE;
  I2C1Handle.I2C_Config_t.I2CDeviceAddress = SLAVE_ADDRS;//address of the slave device
  I2C1Handle.I2C_Config_t.I2CFMDutyCycle = I2C_FM_DUTY_2;
  I2C1Handle.I2C_Config_t.I2CSCLSpeed = I2C_SCL_SPEED_SM;

  I2CInit(&I2C1Handle);
}


int main()
{
	I2C_GPIO_Inits();

	I2C1_Inits();

	I2CPeripheralClockControl(I2C1,ENABLE);
	
	I2CMasterSendData($I2C1Handle,(char*) data,strlen(data),SLAVE_ADDRS)


}
